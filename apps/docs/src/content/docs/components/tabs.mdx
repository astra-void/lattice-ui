## Install

### Global CLI

```bash
lattice add tabs
```

### Monorepo Local Script

```bash
pnpm lattice add tabs
```

## Package

- npm: `@lattice-ui/tabs`
- peer dependencies: `@rbxts/react`, `@rbxts/react-roblox`
- internal dependencies: `@lattice-ui/core`, `@lattice-ui/focus`, `@lattice-ui/layer`

## Primary Exports

- `Tabs`
- `Tabs.Root`
- `Tabs.List`
- `Tabs.Trigger`
- `Tabs.Content`

## Usage Snippet

```tsx
import { Tabs } from "@lattice-ui/tabs";

export function ProfileTabs() {
  return (
    <Tabs.Root defaultValue="overview">
      <Tabs.List>
        <Tabs.Trigger asChild value="overview">
          <textbutton Text="Overview" />
        </Tabs.Trigger>
        <Tabs.Trigger asChild value="activity">
          <textbutton Text="Activity" />
        </Tabs.Trigger>
      </Tabs.List>

      <Tabs.Content asChild value="overview">
        <textlabel Text="Overview content" />
      </Tabs.Content>
      <Tabs.Content asChild value="activity">
        <textlabel Text="Activity content" />
      </Tabs.Content>
    </Tabs.Root>
  );
}
```
