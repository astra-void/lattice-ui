## Install

### Global CLI

```bash
lattice add select
```

### Monorepo Local Script

```bash
pnpm lattice add select
```

## Package

- npm: `@lattice-ui/select`
- peer dependencies: `@rbxts/react`, `@rbxts/react-roblox`
- internal dependencies: `@lattice-ui/core`, `@lattice-ui/focus`, `@lattice-ui/layer`, `@lattice-ui/popper`

## Usage Snippet

```tsx
import { Select } from "@lattice-ui/select";
import { PortalProvider } from "@lattice-ui/layer";

export function RegionSelect({ playerGui }: { playerGui: PlayerGui }) {
  return (
    <PortalProvider container={playerGui}>
      <Select.Root defaultValue="na">
        <Select.Trigger asChild>
          <textbutton Text="Select region" />
        </Select.Trigger>
        <Select.Value asChild placeholder="Pick one">
          <textlabel />
        </Select.Value>

        <Select.Portal>
          <Select.Content asChild>
            <frame>
              <Select.Item asChild textValue="North America" value="na">
                <textbutton Text="North America" />
              </Select.Item>
              <Select.Item asChild textValue="Europe" value="eu">
                <textbutton Text="Europe" />
              </Select.Item>
            </frame>
          </Select.Content>
        </Select.Portal>
      </Select.Root>
    </PortalProvider>
  );
}
```


## Notes

- Single value only in this release.
- Supports controlled/uncontrolled `value` and `open`.
- Content uses dismissable-layer semantics (outside pointer / escape dismiss).
- Item navigation uses roving focus and includes typeahead matching.
- Select.Content can use PortalProvider for robust overlays.
