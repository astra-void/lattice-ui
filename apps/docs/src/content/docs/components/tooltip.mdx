## Install

### Global CLI

```bash
lattice add tooltip
```

### Monorepo Local Script

```bash
pnpm lattice add tooltip
```

## Package

- npm: `@lattice-ui/tooltip`
- peer dependencies: `@rbxts/react`, `@rbxts/react-roblox`
- internal dependencies: `@lattice-ui/core`, `@lattice-ui/layer`, `@lattice-ui/popper`

## Primary Exports

- `Tooltip`
- `Tooltip.Provider`
- `Tooltip.Root`
- `Tooltip.Trigger`
- `Tooltip.Portal`
- `Tooltip.Content`

## Usage Snippet

```tsx
import { PortalProvider } from "@lattice-ui/layer";
import { Tooltip } from "@lattice-ui/tooltip";

export function HintTooltip({ playerGui }: { playerGui: PlayerGui }) {
  return (
    <PortalProvider container={playerGui}>
      <Tooltip.Provider>
        <Tooltip.Root>
          <Tooltip.Trigger asChild>
            <textbutton Text="Hover me" />
          </Tooltip.Trigger>
          <Tooltip.Portal>
            <Tooltip.Content asChild>
              <textlabel Text="Tooltip content" />
            </Tooltip.Content>
          </Tooltip.Portal>
        </Tooltip.Root>
      </Tooltip.Provider>
    </PortalProvider>
  );
}
```


## Notes

- `Tooltip.Provider` coordinates timing behavior across nested or sibling tooltips.
- `Tooltip.Content` is portal-friendly for reliable overlap behavior.
- Tooltip content can be portal-mounted for better overlap behavior.
