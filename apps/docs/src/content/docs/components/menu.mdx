## Install

### Global CLI

```bash
lattice add menu
```

### Monorepo Local Script

```bash
pnpm lattice add menu
```

## Package

- npm: `@lattice-ui/menu`

## Usage Snippet

```tsx
import { Menu } from "@lattice-ui/menu";
import { PortalProvider } from "@lattice-ui/layer";

export function ContextMenu({ playerGui }: { playerGui: PlayerGui }) {
  return (
    <PortalProvider container={playerGui}>
      <Menu.Root>
        <Menu.Trigger asChild>
          <textbutton Text="Open menu" />
        </Menu.Trigger>
        <Menu.Portal>
          <Menu.Content asChild>
            <frame>
              <Menu.Item asChild>
                <textbutton Text="Rename" />
              </Menu.Item>
              <Menu.Item asChild>
                <textbutton Text="Delete" />
              </Menu.Item>
            </frame>
          </Menu.Content>
        </Menu.Portal>
      </Menu.Root>
    </PortalProvider>
  );
}
```

## Notes

- Compose `Menu.Trigger`, `Menu.Content`, and `Menu.Item` for context menus or dropdown menus.
- Grouping and labeling primitives help structure complex menu trees.
- Menu.Content can use PortalProvider for display order control.
